98void IIC_SendByte(unsigned char byt) {
99 unsigned char i;
100 for (i = 0; i < 8; i++) {
101 IIC_SCK(IIC_RESET);
102 IIC_Delay();
103 IIC_SDA((byt & 0x80) != 0);
104 IIC_Delay();
105 IIC_SCK(IIC_SET);
106 byt <<= 1;
107 IIC_Delay();}
108 IIC_SCK(IIC_RESET);}
109//----------------------------------------------------------------------------------------------------
114unsigned char IIC_RecByte(void) {
115 unsigned char i=0, da=0;
116 for (i = 0; i < 8; i++) {
117 IIC_SCK(IIC_SET);
118 IIC_Delay();
119 da <<= 1;
120 if (IIC_SDA(IIC_READ))da |= 1;
121 IIC_SCK(IIC_RESET);
122 IIC_Delay();}
123 return da;}
