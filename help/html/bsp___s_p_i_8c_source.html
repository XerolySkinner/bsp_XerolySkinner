107unsigned char SPI4_Byte(unsigned char temp){
108 unsigned char k=0;
109 unsigned int i=0;
110 unsigned char res=0;
111// CPHA=1
112#if SPI_CPHA // 前沿输出后沿采样
113 SPI4_Delay();
114 for(i=0;i<8;i++){
115 SPI4_SCK_PIN=!SPI_CPOL; // 先沿
116 SPI4_MOSI_PIN=(_crol_(temp,i)&0x80)!=0; // 输出
117 SPI4_Delay();
118 SPI4_SCK_PIN=SPI_CPOL; // 后沿
119 res|=(SPI4_MISO_PIN!=0); // 采样
120 SPI4_Delay();
121 if(i!=7)res<<=1;
122 }
123// CPHA=0
124#else // 前沿采样后沿传输
125 SPI4_MOSI_PIN=(_crol_(temp,0)&0x80)!=0; // 预输出
126 SPI4_Delay();
127 for(i=1;i<9;i++){
128 SPI4_SCK_PIN=!SPI_CPOL; // 先沿
129 res|=(SPI4_MISO_PIN!=0); // 采样
130 SPI4_Delay();
131 if(i!=8)res<<=1;
132 SPI4_SCK_PIN=SPI_CPOL; // 后沿
133 SPI4_MOSI_PIN=(_crol_(temp,i)&0x80)!=0; // 输出
134 SPI4_Delay();
135 }
136#endif
137 return res;
138 }
